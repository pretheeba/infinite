--1. Write a T-SQL Program to find the factorial of a given number.
create or alter proc factorial
@n int
 
as
begin
declare @i int=1
declare @x int=1
  while @i<=@n 
   begin
      set @x=@x*@i
      set @i=@i+1
   end
end
print 'factorial of given number is:' + cast(@x as varchar(10))
 
factorial 4
